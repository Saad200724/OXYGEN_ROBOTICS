      activeFilter = category;

    // Interactive search system
    function searchMembers() {
      applyFilters();
    }

    function applyFilters() {
      const searchVal = document.getElementById('searchBar').value.toLowerCase();
      
      const filtered = committeeData.filter(member => {
        const matchesCategory = (activeFilter === 'all' || member.category === activeFilter);
        const matchesSearch = member.name.toLowerCase().includes(searchVal) || 
                              member.role.toLowerCase().includes(searchVal) || 
                              member.dept.toLowerCase().includes(searchVal);
        return matchesCategory && matchesSearch;
      });

      renderCards(filtered);
    }

            </div>
          </div>
        `;
      }

      content.innerHTML = `
        <div class="flex flex-col md:flex-row gap-6 items-center md:items-start">
          <!-- Left Visual Avatar Column -->
          <div class="w-32 h-32 shrink-0 rounded-full p-[3px] bg-gradient-to-tr ${member.colorTheme} shadow-[0_0_20px_rgba(217,70,239,0.15)]">
            <div class="w-full h-full bg-neutral-950 rounded-full flex items-center justify-center overflow-hidden">
              <svg class="w-20 h-20 text-neutral-400" viewBox="0 0 100 100" fill="none">

            <div class="flex flex-wrap gap-2 justify-center md:justify-start items-center mt-3">
              <span class="text-[11px] font-mono font-bold tracking-wider uppercase px-2.5 py-1 rounded-md border ${accentBg} ${accentText}">

    function closeInspector() {
      const modal = document.getElementById('inspectorModal');
      const modalCard = document.getElementById('modalCard');

      modal.classList.add('opacity-0', 'pointer-events-none');
      modalCard.classList.remove('scale-100');
      modalCard.classList.add('scale-95');
    }

    // Global Close Listener (clicking outside the card)
    window.addEventListener('click', (e) => {
      const modal = document.getElementById('inspectorModal');
      if (e.target === modal) {
        closeInspector();
      }
    });

      renderCards();
